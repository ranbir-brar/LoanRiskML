\section{Exploratory Data Analysis}

% --- Slide 1: Target Imbalance & Data Scope ---
\begin{frame}{Target Imbalance \& Data Scope}
  \begin{columns}[T]

    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \item Only about 15\% of known outcomes are defaults.
              With that kind of imbalance, raw accuracy is
              misleading, so we use F1 and AUC instead.
        \vspace{0.5em}
        \item About 8\% of applications are still
              \texttt{ongoing} with no final label, so they're
              excluded from training.
        \vspace{0.5em}
        \item The chart shows how small the default class is
              relative to repaid; any model that just predicts
              ``repaid'' would look 85\% accurate.
      \end{itemize}
    \end{column}

    \begin{column}{0.55\textwidth}
      \includegraphics[width=\linewidth]{plot_02_class_distribution.png}
    \end{column}

  \end{columns}
\end{frame}

% --- Slide 2: Missing Data as a Predictive Signal ---
\begin{frame}{Missing Data as a Predictive Signal}
  \begin{columns}[T]

    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \item 15\% of applications are missing
              \texttt{documented\_monthly\_income}.
        \vspace{0.6em}
        \item Missingness is \alert{not random}; applicants
              without documentation default at a much higher
              rate.
        \vspace{0.6em}
        \item So missing docs isn't just a data quality
              problem; it's a genuine risk signal.
      \end{itemize}
    \end{column}

    \begin{column}{0.55\textwidth}
      \includegraphics[width=\linewidth]{plot_01_missingness.png}
    \end{column}

  \end{columns}
\end{frame}

% --- Slide 3: Behavior Signals Over Raw Values ---
\begin{frame}{Behaviour Signals Over Raw Values}
  \begin{columns}[T]

    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \item When stated income is more than $2\times$
              documented income, default rates spike.
              Misrepresentation is a strong risk signal.
        \vspace{0.6em}
        \item Default rate jumps from \textasciitilde17\% to
              \textasciitilde42\% once the loan is more than
              30\% of monthly income.
        \vspace{0.6em}
        \item The raw dollar amount matters less than how
              the income \textit{relates} to the loan size.
      \end{itemize}
    \end{column}

    \begin{column}{0.55\textwidth}
      \includegraphics[width=\linewidth]{plot_03_income_misrepresentation.png}
    \end{column}

  \end{columns}
\end{frame}

\section{Model Choice}

% --- Slide 4: Why XGBoost? ---
\begin{frame}{Why XGBoost?}
  \begin{columns}[T]

    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \item Picks up non-linear interactions between
              features --- logistic regression can't do this.
        \vspace{0.6em}
        \item \texttt{scale\_pos\_weight} deals with class
              imbalance natively, no resampling needed.
        \vspace{0.6em}
        \item Threshold tuned for F1 on defaults,
              not the naive 0.5 cutoff.
      \end{itemize}
    \end{column}

    \begin{column}{0.55\textwidth}
      \includegraphics[width=\linewidth]{plot_threshold.png}
    \end{column}

  \end{columns}
\end{frame}

% --- Slide 8: The Deploy-Tomorrow Trade-off ---
\begin{frame}{The Deploy-Tomorrow Trade-off}
  \begin{columns}[T]

    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \item XGBoost catches \textbf{+37} more defaults per
              368 applications vs.\ the baseline (33\% $\to$ 67\%
              recall).
        \vspace{0.5em}
        \item Trade-off: \textbf{+69} additional good applicants
              wrongly denied (8.5\% $\to$ 35.1\% FPR).
        \vspace{0.5em}
        \item \emph{Who decides?} The founders, weighing the
              cost of a missed default (full loan loss) against
              the cost of losing a customer (CAC). The threshold
              is a dial, not a fixed number.
      \end{itemize}
    \end{column}

\section{Fairness \& AI Ethics}

% --- Slide 9: Algorithmic Fairness ---
\begin{frame}{Algorithmic Fairness \& the Ethics of Automated Decisions}
  \begin{columns}[T]

    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \item The rule-based system penalises self-employed
              applicants even though their \alert{actual default
              rates} are similar to employed applicants. That's
              textbook proxy discrimination.
        \vspace{0.4em}
        \item XGBoost learns from outcomes, not labels, so
              approval rates track real risk instead of
              employment category.
        \vspace{0.4em}
        \item Tension: keeping \texttt{employment\_status}
              risks disparate-impact liability even if the model
              uses it correctly. Removing it may increase defaults.
              \emph{Neither choice is cost-free.}
      \end{itemize}
    \end{column}

    \begin{column}{0.55\textwidth}
      \includegraphics[width=\linewidth]{plot_fairness.png}

\end{frame}

\section{Production}

% --- Slide 10: If This Went Live Tomorrow... ---
\begin{frame}{If This Went Live Tomorrow\ldots}
  \begin{itemize}
    \item \textbf{Model drift:} a recession doubling default rates
          would silently degrade predictions. The rule-based system
          at least degrades \textit{transparently}.
    \vspace{0.6em}
    \item \textbf{Distribution monitoring:} feature distributions
          (e.g., \texttt{loan\_to\_income} under inflation) need
          to be tracked continuously. The static rule-based system
          doesn't need any of that.
    \vspace{0.6em}
    \item \textbf{Regulatory audits:} covered. The local explainer
          sits alongside the prediction API and generates
          human-readable denial reasons per applicant.
  \end{itemize}
\end{frame}
